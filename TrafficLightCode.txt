
void setup()
{
  pinMode(2, OUTPUT); // RED LED
  pinMode(3, OUTPUT); // YELLOW LED
  pinMode(4, OUTPUT); // GREEN LED
}

void loop()
{
  digitalWrite(2, HIGH);
  delay(5000); // Wait for 5000 millisecond(s)
  digitalWrite(2, LOW);
  
  digitalWrite(3, HIGH);
  delay(2000); // Wait for 2000 millisecond(s)
  digitalWrite(3, LOW);

  
  digitalWrite(4, HIGH);
  delay(5000); // Wait for 5000 millisecond(s)
  digitalWrite(4, LOW);
 
}
